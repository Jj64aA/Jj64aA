 def fibonacci(n):
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
    else:
        sequence = [0, 1]
        a, b = 0, 1
        for _ in range(2, n):
            c = a + b
            sequence.append(c)
            a, b = b, c
        return sequence
